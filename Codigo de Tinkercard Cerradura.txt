#include <Keypad.h>
#include <LiquidCrystal.h>

//inicializamos cartel y declamos tama√±o de teclado
LiquidCrystal cartel(13, 8, A0, A1, A2, A3);

const byte FILAS = 4;
const byte COLUMNAS = 4;

//declaramos array de caracteres de teclado
char caracteres[4][4] = {
  {'1','2','3','A'},
  {'4','5','6','B'},
  {'7','8','9','C'},
  {'*','0','#','D'},  
};

//declaramos entradas de pines y mapeamos funcion
byte pinesFilas[] = {7,6,5,4};
byte pinesColumnas[] = {3,2,1,0};

Keypad teclado = Keypad( makeKeymap(caracteres)
                ,pinesFilas, pinesColumnas,
                FILAS, COLUMNAS);

void setup()
{
  pinMode(8, OUTPUT);
  pinMode(9, OUTPUT);
  pinMode(10, OUTPUT);
  pinMode(13, OUTPUT); 
  cartel.begin(16,2);
  Serial.begin(9600);
}

void loop()
{  
  //seteamos cursor
  cartel.setCursor(0,0);
  
  //escribimos el mensaje
  cartel.write("Password:");
  
  //inicializamos variable y asignacion    
  char boton = teclado.getKey();
  
    if(boton != NO_KEY){
  	Serial.println(boton);      
    cartel.print(boton);
  }
  
  
  
  //Leemos el ingreso y prendemos los leds
  if(boton == '2'){
    digitalWrite(9, HIGH);
    digitalWrite(10, LOW);
  }
  else{
    if(boton == '7'){
      digitalWrite(9, LOW);
      digitalWrite(10, HIGH);
    }
    else{
      //reinciamos con leds con asterisco
      if(boton == '*'){
        digitalWrite(9, LOW);
        digitalWrite(10, LOW);
      }
    }
  }
  
  delay(25);
}
